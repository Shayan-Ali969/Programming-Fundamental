#include <stdio.h>

int main() {
    int arr[100], size, i;
    int first, second;

    printf("Enter number of elements: ");
    scanf("%d", &size);

    printf("Enter %d numbers:\n", size);
    for (i = 0; i < size; i++) {
        scanf("%d", &arr[i]);
    }

    first = second = arr[0];

    for (i = 1; i < size; i++) {
        if (arr[i] > first) {
            second = first;
            first = arr[i];
        } else if (arr[i] > second && arr[i] != first) {
            second = arr[i];
        }
    }

    printf("Two largest numbers: %d and %d\n", first, second);

    return 0;
}
